7-55 到底有多二（15 分）
一个整数“犯二的程度”定义为该数字中包含2的个数与其位数的比值。如果这个数是负数，则程度增加0.5倍；如果还是个偶数，则再增加1倍。例如数字-13142223336是个11位数，其中有3个2，并且是负数，也是偶数，则它的犯二程度计算为：3/11×1.5×2×100%，约为81.82%。本题就请你计算一个给定整数到底有多二。

输入格式：
输入第一行给出一个不超过50位的整数N。

输出格式：
在一行中输出N犯二的程度，保留小数点后两位。

输入样例：
-13142223336
输出样例：
81.82%


#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    int c=0, flag=0;
    double sum=1;
    getline(cin, s);
    for(int i=0;i<s.length();i++){
        if(s[i]=='-'&&i==0){
            sum = sum * 1.5;
            flag=1;
        }
        if(i==s.length()-1){
            if((s[i]-'0')%2==0){
                sum = sum * 2;
            }
        }
        if(s[i]=='2'){
            c++;
        }
    }
    if(flag==0){
        sum = sum * ((double)c/s.length()) * 100;
    }
    else{
        sum = sum * ((double)c/(s.length()-1)) * 100;
    }
    printf("%.2lf", sum);
    cout<<"%"<<endl;
    return 0;
}
